f(t)=∑a=1nR(a)ev(a,t)if(t)=∑_{a=1}^{n}{R(a)\mathrm{e}^{v(a,t)i}} f(t)=a=1∑n​R(a)ev(a,t)i

Clear["Global`*"];
n = 20;
pt = {{-8., 0.}, 1};
R[a_] = 2/a;
v[a_] = 2 a t;
tiaozheng[{{x_, y_}, a_}] := {{x, y}, R[a]};
f[pt_] := Block[{pts, a},
   pts = pt[[1]]; a = pt[[2]];
   Return[{TranslationTransform[{R[a] Cos[v[a]], R[a] Sin[v[a]]}][pts], a + 1}];
   ];
pts = Flatten[Drop[#, -1] & /@ NestList[f, pt, n], 1];
pts1 = pts[[-1, -1]] /. t -> (t + u);
ListAnimate[ParallelTable[
  Show[
   Graphics@Point@pts,
   Graphics@Line@pts,
   Graphics /@ Circle @@@ (tiaozheng /@ NestList[f, pt, n][[;; -2]]),
   Graphics@Line[{pts[[-1]], {0, pts[[-1]][[-1]]}}],
   ListLinePlot[Table[{u, pts1}, {u, 0, t, 0.01}]],
   Axes -> True, PlotRange -> {{-11., 6.}, {-5., 5.}}
   ], {t, 12.3, 6.3, -0.04}]]

img = Binarize[img~ColorConvert~"Grayscale"~ImageResize~500~Blur~3]
pts = DeleteDuplicates@
   Cases[Normal@
      ListContourPlot[Reverse@ImageData[img], 
       Contours -> {0.5}], _Line, -1][[1, 1]];
center = Mean@MinMax[pts] & /@ Transpose@pts;
pts = # - center & /@ pts[[;; ;; 20]];
elephantPlot = ListPlot[pts, AspectRatio -> Automatic]
SetAttributes[toPt, Listable]
toPt[z_] := ComplexExpand[{Re@z, Im@z}] // Chop;
cf = Compile[{{z, _Complex, 1}}, 
   Module[{n = Length@z}, 
    1/n*Table[Sum[z[[k]]*Exp[-I*i*k*2 Pi/n], {k, 1, n}], {i, -m, m}]]];
z = pts[[All, 1]] + I*pts[[All, 2]];
m = 50;
cn = cf[z];
{f[t_], g[t_]} = 
  Sum[cn[[j]]*Exp[I*(j - m - 1)*t], {j, 1, 2 m + 1}] // toPt;
ParametricPlot[{f[t], g[t]}, {t, 0, 2 Pi}, AspectRatio -> Automatic]
r = Abs /@ cn;
theta = Arg /@ cn;
index = {m + 1}~Join~
   Riffle[Range[m + 2, 2 m + 1], Reverse[Range[1, m]]];
p[t_] = Accumulate@Table[cn[[j]]*Exp[I*(j - m - 1)*t], {j, index}] // 
   toPt;
circles[t_] = 
  Table[Circle[p[t][[i]], r[[index[[i]]]]], {i, 1, 2 m + 1}];
anims = ParallelTable[
   ParametricPlot[{f[s], g[s]}, {s, 0, t}, AspectRatio -> Automatic, 
    Epilog -> {circles[t][[2 ;;]], Line[p[t]], Point[p[t]]}, 
    PlotRange -> {{-400, 400}, {-400, 200}}, ImageSize -> 500], {t, 
    Subdivide[0.1, 4 Pi, 1000]}];
